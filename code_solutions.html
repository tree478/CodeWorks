Lesson 2: Hexagon Spiral

								
import turtle

window = turtle.Screen()
window.title("Spiral")
window.setup(300, 200)
window.bgcolor('black')

t = turtle.Turtle()
t.pencolor('white')
t.speed(0)
t.width(3)

for x in range(50):
	t.forward(x + 1)
	t.left(66)

turtle.mainloop()
